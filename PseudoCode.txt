void CalculateHealth()
{
	if (isAverage)
	{
		int totalNumber;
		for (int i = 1, i <= classes[playerClass], i++)
		{
			totalNumber += i;
		}
		int average = Convert.Int32(totalNumber/classes[playerClass]);
		hp += average * playerLevel;
	}
	else
	{
		int totalNumber;
		for (int i = 0, i < playerLevel, i++)
		{
			totalNumber += Random.Range(1, classes[playerClass] + 1)
		}
		hp += totalNumber
	}
	switch (conScore)
	{
		case 1:
			hp += -5 * playerLevel;
			break;
		case 2:
		case 3:
			hp += -4 * playerLevel;
			break;
		...
		case 30:
			hp += 10 * playerLevel;
	}
	if (playerFeat.Contains("Tough"))
	{
		hp += 2 * playerLevel
	}
	if (playerFeat.Contains("Stout"))
	{
		hp += playerLevel	
	}
	if (playerRace == "Dwarf")
	{
		hp += playerLevel * 2
	}
	else if (playerRace == "Orc" || playerRace == "Goliath")
	{
		hp += playerLevel
	}
}
